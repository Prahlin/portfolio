     3	import { useEffect, useState } from "react";
     4	import { ArrowRight } from "lucide-react";
     5	
     6	const projectLinks = [
     7	  { href: "#cinerific", label: "View 'Cinerific'" },
     8	  { href: "#alla-vostra", label: "View 'Alla Vostra'" },
     9	  { href: "#credit-king", label: "View 'Credit King'" },

    11	
    12	export function ProjectCarouselButton() {
    13	  const [activeIndex, setActiveIndex] = useState(1);
    14	  const activeProject = projectLinks[activeIndex];
    15	
    16	  useEffect(() => {
    17	    const interval = window.setInterval(() => {
    18	      setActiveIndex((currentIndex) => (currentIndex + 1) % projectLinks.length);
    19	    }, 5000);
    20	
    21	    return () => window.clearInterval(interval);
    22	  }, []);
    23	
    24	  return (
    25	    <a
    26	      className="button button-primary carousel-button"
    27	      href={activeProject.href}
    28	    >
    29	      <span className="carousel-button-window" aria-live="polite">
    30	        <span className="carousel-button-label" key={activeProject.href}>
    31	          {activeProject.label}
    32	        </span>
    33	      </span>

     1	# Prahlin Portfolio Site
     2	
     3	Next.js + Tailwind CSS portfolio start page for Prahlin.
     4	
     5	## What is included
     6	
     7	- `app/page.tsx` - the homepage/start page.
     8	- `app/globals.css` - the green/black visual system and responsive layout.
     9	- `public/images/profile-placeholder.png` - generated AI profile placeholder; replace this with your own square photo later using the same filename.
    10	- `public/images/start-page-mockup-reference.png` - the original generated mockup reference.
    11	- `public/favicon.svg` - simple green/black monogram favicon.
    12	
    13	## Local development
    14	
    15	```sh
    16	npm install
    17	npm run dev
    18	```
    19	
    20	Open `http://localhost:3000`.
    21	
    22	## Production build
    23	
    24	For Vercel, Netlify, or another Next.js host:
    25	
    26	```sh
    27	npm run build
    28	```
    29	
    30	For a simple static host:
    31	
    32	```sh
    33	npm run build:static
    34	```
    35	
    36	Then upload the generated `out/` folder.
    37	
    38	## Replace the profile photo
    39	
    40	Put your real square profile photo at:
    41	
    42	```txt
    43	public/images/profile-placeholder.png
    44	```
    45	
    46	Keep the filename the same and rebuild.

     1	# Deployment Notes
     2	
     3	## Best Option: Vercel
     4	
     5	1. Upload this `portfolio-site` folder to a GitHub repo.
     6	2. Import the repo in Vercel.
     7	3. Use the default Next.js settings.
     8	4. Build command: `npm run build`.
     9	
    10	## Netlify
    11	
    12	1. Upload this folder to a GitHub repo.
    13	2. Connect the repo in Netlify.
    14	3. Build command: `npm run build`.
    15	4. Publish directory: `.next`.
    16	5. Enable Netlify's Next.js runtime if prompted.
    17	
    18	## Simple Static Hosting
    19	
    20	Use this when the host only accepts HTML/CSS/JS files:
    21	
    22	```sh
    23	npm install
    24	npm run build:static
    25	```
    26	
    27	Upload the contents of:
    28	
    29	```txt
    30	out/
    31	```
    32	
    33	This mode does not need a running Node.js server after upload.
    34	
    35	## Contact Button
    36	
    37	The contact button currently uses:
    38	
    39	```txt
    40	mailto:hello@example.com
    41	```
    42	
    43	Replace `hello@example.com` in `app/page.tsx` with your real email address.

   107	print_file_inventory_line() {
   108	  local file="$1"
   109	  local size
   110	  local modified
   111	
   112	  size="$(stat -f "%z" "$file" 2>/dev/null || stat -c "%s" "$file" 2>/dev/null || echo "unknown")"
   113	  modified="$(stat -f "%Sm" -t "%Y-%m-%d %H:%M:%S" "$file" 2>/dev/null || stat -c "%y" "$file" 2>/dev/null || echo "unknown")"
   114	  printf '%s | %s bytes | modified %s\n' "${file#$PROJECT_DIR/}" "$size" "$modified" >> "$OUTPUT_FILE"
   115	}

   117	print_dir_inventory() {
   118	  local title="$1"
   119	  local dir="$2"

   121	  print_inventory_header "$title"

   123	  if [ ! -d "$dir" ]; then

   126	  fi

   128	  find "$dir" -type f \
   129	    ! -name ".DS_Store" \
   130	    ! -name "*_PROJECT_SNAPSHOT.txt" \
   131	    | sort | while IFS= read -r file; do
